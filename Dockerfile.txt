# Stato: COMPLETO (Yii2 + Imagick + Fix Permessi DigitalOcean 2026)
FROM php:8.2-apache

# 1. Installazione dipendenze e ImageMagick
RUN apt-get update && apt-get install -y \
    libmagickwand-dev --no-install-recommends \
    fonts-dejavu-core \
    fonts-dejavu \
    fontconfig \
    libicu-dev \
    libzip-dev \
    libfreetype6-dev \
    libfontconfig1-dev \
    && pecl install imagick \
    && docker-php-ext-enable imagick \
    && docker-php-ext-install intl zip pdo_mysql \
    && rm -rf /var/lib/apt/lists/* \
    && fc-cache -f -v

# 2. Abilitiamo il modulo rewrite di Apache
RUN a2enmod rewrite

# 3. Impostazione cartella di lavoro
WORKDIR /app

# 4. Configurazione Apache per Yii2
ENV APACHE_DOCUMENT_ROOT /app/frontend/web
RUN sed -ri -e 's!/var/www/html!${APACHE_DOCUMENT_ROOT}!g' /etc/apache2/sites-available/000-default.conf
RUN sed -ri -e 's!/var/www/!${APACHE_DOCUMENT_ROOT}!g' /etc/apache2/conf-available/*.conf

# 5. Copia del codice (Assicurati che i font siano in frontend/web/fonts)
COPY . /app

# 6. FIX PERMESSI DEFINITIVO
# Creiamo prima le cartelle necessarie, poi cambiamo il proprietario a tutto
RUN mkdir -p /app/frontend/runtime \
    /app/frontend/web/assets \
    /app/frontend/web/uploads/targets \
    /app/backend/runtime \
    /app/backend/web/assets \
    && chown -R www-data:www-data /app \
    && chmod -R 775 /app/frontend/runtime \
    && chmod -R 775 /app/frontend/web/assets \
    && chmod -R 775 /app/frontend/web/uploads

# 7. FIX ENTRYPOINT
RUN chmod +x /usr/local/bin/docker-php-entrypoint

# 8. CONFIGURAZIONE SPECIFICA PERMESSI APACHE (Risolve AH01630)
RUN echo "<Directory /app/frontend/web/>\n\
    Options Indexes FollowSymLinks\n\
    AllowOverride All\n\
    Require all granted\n\
</Directory>" > /etc/apache2/conf-available/yii2.conf \
    && a2enconf yii2
EXPOSE 80

# Usiamo l'utente www-data per far girare Apache in sicurezza se necessario
# Ma per ora lasciamo il default per non complicare la connessione alle porte basse
CMD ["apache2-foreground"]
